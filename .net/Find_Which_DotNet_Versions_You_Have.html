Find which .net versions you have

On Windows 10 or 11

reg query "HKLM\SOFTWARE\Microsoft\Net Framework Setup\NDP" /s | 
	? { $_ -like "*version*" } | 
	% { ($_ -split " ") | 
	? { $_ -match "[0-9.]+" } | 
			% { $_ } 
		} |
	sort -desc |
	unique
